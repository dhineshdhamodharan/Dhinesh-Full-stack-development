class Circle
{
    constructor(radius=1.0,color="red")
    {
    this.radius=radius;
    this.color=color;
    }
    getRadius(circumference)
    {
     var r=circumference/2*3.14;
      return r;
    }
    setRadius(radius)
    {
      this.radius=radius;
    }
    getColor()
    {
      return this.color;
    }
    setColor(color)
    {
      this.color=color;
    }
    getArea()
    {
      var area=3.14*this.radius*this.radius;
      return area;
    }
    getCircumference()
    {
      var circumference=2*3.14*this.radius;
      return circumference
    }
  }
